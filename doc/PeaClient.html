    Params:

    partition_name

    DNS name (name of the partition) of the remote AS/400 server.

    port

    Port used for the data exchange between the client and the server.

    username

    Username of the AS/400 profile used for connexion.

    password

    Password of the AS/400 profile used for connexion.

    id_client

    ID of the client account on the DIPS website.

    online_version

    Set to true if you want to use the online version of Peasys (<see cref=“dips400.com/docs/connexion”/>).

    retrieve_statistics

    Set to true if you want the statistics of the license key use to be collect.

    throws a PeaInvalidCredentialsError if credentials are empty. throws a PeaConnexionError if an error occured during the connexion process. throws a PeaInvalidCredentialsError if IBMi credentials are invalid.

def initialize(partition_name, port, username, password, id_client, online_version, retrieve_statistics)

        @tcp_client = TCPSocket.new partition_name, port

        throw PeaConnexionError.new(error.full_message)
    else
        login = username.ljust(10, " ") + password.ljust(10, " ")
        @tcp_client.send(login, 0)
        @connexion_status = @tcp_client.read(1).to_i

            throw PeaConnexionError.new("Product expired")
        else
            throw PeaConnexionError.new("Exception during connexion process, contact us for more informations")
        end
    end

def execute_alter(query)
    if query.empty?
        throw PeaInvalidSyntaxQueryError.new("Query should not be either null or empty")
    end
    if !query.upcase.start_with?("ALTER")
        throw PeaInvalidSyntaxQueryError.new("Query should start with the ALTER SQL keyword")
    end

    nb_row, sql_state, sql_message = modify_table(query)
    
    return PeaAlterResponse.new(sql_message.eql?("00000"), sql_message, sql_state, null)
end

def execute_command(command)
    data = retrieve_data("exas" + command + @@end_pack)
    description_offset = 112
    has_succeeded = true
    result = Array.new()

    /C[A-Z]{2}[0-9]{4}/.match(data) do |m|
        
        if !m[0].start_with?("CPI")
            description = data[m.offset(0)[0] + description_offset..(data.length - (m.offset(0)[0] + description_offset))]
            description = description[0..description.index(".")]
            description = description.gsub(/[^a-zA-Z0-9 ._'*:-]/, '')
        end

        result.append(m[0] + " " + description)

        if m[0].start_with?("CPF")
            has_succeeded = false
        end
    end
    return PeaCommandResponse.new(has_succeeded, result)
end

def execute_update(query)
    if query.empty?
        throw PeaInvalidSyntaxQueryError.new("Query should not be either null or empty")
    end
    if !query.upcase.start_with?("UPDATE")
        throw PeaInvalidSyntaxQueryError.new("Query should start with the UPDATE SQL keyword")

    nb_row, sql_state, sql_message = modify_table(query)
    has_succeeded = sql_state.eql?("00000") || sql_state.eql?("01504")
    
    return PeaUpdateResponse.new(has_succeeded, sql_message, sql_state, nb_row)
